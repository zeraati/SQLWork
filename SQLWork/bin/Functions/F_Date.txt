CREATE FUNCTION [dbo].[F_Date] 
(
@date VARCHAR(20)
)
RETURNS VARCHAR(10)
AS
BEGIN
	DECLARE @Result VARCHAR(10)
	DECLARE @str	VARCHAR(10)

	IF CHARINDEX('/',@date)>=4 OR CHARINDEX('-',@date)>=4 OR CHARINDEX('.',@date)>=4
		BEGIN

		RETURN @Result
		END
    ELSE
		BEGIN
		RETURN @Result
		end

	RETURN @Result
END
